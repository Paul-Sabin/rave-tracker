3. Add JavaScript function for button tracking (in the `<script>` section):

```javascript
async function trackArtistFromButton(btn) {
    if (btn.classList.contains('tracked')) return;

    const artistId = btn.dataset.artistId;
    const artistName = btn.dataset.artistName;

    btn.disabled = true;
    btn.querySelector('.artist-btn-action').textContent = '...';

    try {
        const res = await fetch('/api/rules/add', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                rule_type: 'artist',
                target_id: artistId,
                target_name: artistName
            })
        });
        const data = await res.json();

        if (data.success) {
            btn.classList.add('tracked');
            showToast(`Now tracking ${artistName}`);
        } else if (data.error && data.error.includes('Already tracking')) {
            btn.classList.add('tracked');
        } else {
            btn.querySelector('.artist-btn-action').textContent = 'Error';
            showToast(data.error || 'Failed to add rule', true);
        }
    } catch (err) {
        btn.querySelector('.artist-btn-action').textContent = '+ Track';
        showToast('Failed to add rule', true);
    }

    btn.disabled = false;
}
```
